```python
def solution(n, cores):
    if n <= len(cores):
        return n

    left, right = 0, max(cores) * n

    while left < right:
        mid = (left + right) // 2
        completed = sum(mid // core for core in cores)
        
        if completed >= n - len(cores):
            right = mid
        else:
            left = mid + 1

    time = left
    remaining_work = n - len(cores) - sum((time - 1) // core for core in cores)

    for i, core in enumerate(cores):
        if time % core == 0:
            remaining_work -= 1
            if remaining_work == 0:
                return i + 1
```
